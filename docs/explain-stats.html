# Load libraries
library(tidyverse)
library(lubridate)

# Process data
sales <- read_csv("sales.csv")
sales$date <- as.Date(sales$date)
sales$month <- floor_date(sales$date, "month")

# Calculate summaries
monthly_stats <- sales %>%
  group_by(month, category) %>%
  summarize(
    total = sum(amount),
    average = mean(amount),
    count = n()
  )

# Create plots
ggplot(monthly_stats, aes(x = month, y = total)) +
  geom_line() +
  facet_wrap(~category)

# Export results
write_csv(monthly_stats, "monthly_summary.csv")
